{{/* Inline SVG helper: injects icon-specific class for styling */}} {{- $svg := . -}} {{- $class := printf "%s-icon" $svg -}} {{- $match := `]*)>([\s\S]*?)` -}} {{- $file := printf "/assets/images/%s.svg" $svg | readFile -}} {{- $clean := replaceRE `(?s)^(?:\s*<\?xml[^>]*>\s*)?(?:\s*\s*)*` "" $file -}} {{- $replaceWith := printf `` $class -}} {{- return (replaceRE $match $replaceWith $clean | safeHTML) -}}
