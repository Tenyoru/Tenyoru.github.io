{{/* Inline SVG helper: injects icon-specific class for styling */}} {{- $svg := . -}} {{- $class := printf "%s-icon" $svg -}} {{- $match := `]*)>([\s\S]*?)` -}} {{- $file := printf "/assets/images/%s.svg" $svg | readFile -}} {{- $replaceWith := printf `` $class -}} {{- return (replaceRE $match $replaceWith $file | safeHTML) -}}
